#include <stdio.h>

int fibonacci(int n) {
    if (n <= 1)
        return n;
    else
        return fibonacci(n - 1) + fibonacci(n - 2);
}

int main() {
    int n;
    printf("Fibonachi sonlarining tartib raqamini kiriting: ");
    scanf("%d", &n);

    printf("Fibonachi soni: %d\n", fibonacci(n));

    return 0;
}
